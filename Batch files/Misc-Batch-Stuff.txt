rem robocopy parameter breakdown:** COPY OPTIONS ** [/zb] = restartable mode, if acess is denied switches to backup mode. Allows robocopy to restart where it was interupted and also overwrite permissions if needed.	[/e] = Copies subdirectories & include empty ones.	[/xx] = exclude extra. If used in conjunction with /Purge or /Mir, will prevent any files being deleted from the destination. Also prevents extra files from being listed in log.
rem													[/mt:] = Creates multi-threaded copies with n threads. 8 is standard.	[/r] = Retries in case of errors.	[/W] = wait time between retries.
rem								 ** SELECTION ** [/xo] = Source files older than destination are excluded.	[/is] & [/it] = Include the same & tweaked files. USed in some instances to ensure it "overwrites" any existing "settings".
rem								 **	LOGGING ** [/v] = Verbose, shows all skipped files too.	[/np] = Progress won't be displayed.	[/log+:] = Append new entries to LogFile.
robocopy "<SOURCE>" "<DESTINATION>" /zb /e /mt:8 /r:3 /w:5 /v /np /log+:"%LogFile%"

rem find the username of the one logged in, not who's running the script.
for /f "tokens=1,2 delims=\" %%r in ('powershell -command "(Get-WMIObject -class Win32_ComputerSystem | select username)"') do set "usrname=%%s"
rem Here we go! This gets the [full path] for local machine and our user, correct SID
for /f "delims=" %%g in ('powershell -command "(ls 'hklm:software/microsoft/windows nt/currentversion/profilelist' | ? {$_.getvalue('ProfileImagePath') -match '%usrname%'} | select Name)"') do set "bread=%%g"
rem Cutting of and keeping only the SID from our previous result.
for /f "tokens=6* delims=\" %%i in ("%bread%") do set "R_USER=%%j"

					LOG FILE CREATION
rem Create Log file
rem Tweak path variable for desired log location, default is "C:\Users\USERNAME\<logfilename>"
set "LogFilePath=%userprofile%\"
for /f "tokens=2 delims==" %%a in ('wmic OS Get localdatetime /value') do set "dt=%%a"
set "YY=%dt:~2,2%" & set "YYYY=%dt:~0,4%" & set "MM=%dt:~4,2%" & set "DD=%dt:~6,2%"
set "HH=%dt:~8,2%" & set "Min=%dt:~10,2%" & set "Sec=%dt:~12,2%"
set "fullstamp=%YYYY%-%MM%-%DD%_%HH%%Min%-%Sec%"
set "LogFile=%LogFilePath%%fullstamp%.log"
